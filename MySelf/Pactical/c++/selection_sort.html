        #include<iostream>
        using namespace std;
        void selection_sort(int array[],int size)
        {
            int i,j,x;
            for(i=0;i<size;i++)
            {
                x=i;
                for(j=i+1;j<size;j++)
                {
                    if(array[j]<array[x])
                    {
                        x=j;
                    }
                }
                swap(array[i],array[x]);
            }
        }
        int main()
        {
            int size,array[30],i,x;
            cout<<"Enter number of element: ";
            cin>>size;
            cout<<"Enter elements: ";
            for(i=0;i<size;i++)
            {
                cin>>array[i];
            }
            cout<<"Before Sorting elements: ";
            for(i=0;i<size;i++)
            {
                cout<<array[i]<<"\t";
            }
            selection_sort(array,size);
            cout<<"\nAfter Sorting elements: ";
            for(i=0;i<size;i++)
            {
                cout<<array[i]<<"\t";
            }
            return 0;
        }
