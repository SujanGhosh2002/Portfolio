        #include<iostream>
        using namespace std;
        void insertion_sort(int array[],int size)
        {
            int i,j,x;
            for(i=1;i<size;i++)
            {
                x=array[i];
                for(j=i-1;(j>=0 && array[j]>x);j--)
                {
                    array[j+1]=array[j];
                }
                array[j+1]=x;
            }
        }
        int main()
        {
            int size,array[30],i;
            cout<<"Enter number of element: ";
            cin>>size;
            cout<<"Enter elements: ";
            for(i=0;i<size;i++)
            {
                cin>>array[i];
            }
            cout<<"Before Sorting elements: ";
            for(i=0;i<size;i++)
            {
                cout<<array[i]<<"\t";
            }
            insertion_sort(array,size);
            cout<<"\nAfter Sorting elements: ";
            for(i=0;i<size;i++)
            {
                cout<<array[i]<<"\t";
            }
            return 0;
        }
